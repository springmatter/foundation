# SmMore

A dropdown icon to display more options or information.

<Props of={SmMore} />

## Standard More

Specify a width class for the SmMore component to best fit the text in the options.

<Preview>
  <Story name="standard" height="300px">
    {{
      components: { SmMore },
      data() {
        return {
          event: "som"
        };
      },
      methods: {
        setEvent(str) {
          this.event = str;
        }
      },
      template: `
        <div>
            <div class="ma2 pa2 absolute">
                <SmMore  class="mv2 w5"
                    :options="['Edit', 'Delete', 'Change Location']"
                    v-on:edit="setEvent('edit')"
                    v-on:delete="{action('delete')}"
                    v-on:change_location="event='change_location'"/>
                <SmMore class="mv2 w-100"
                    :options="['Monday Tuesday Wednesday Thursday Friday Saturday Sunday', 'None']"
                    />
            </div>
        </div>
        `
    }}
  </Story>
</Preview>
