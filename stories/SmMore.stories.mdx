# SmMore

A dropdown icon to display more options or information.

<Props of={SmMore} />

## Standard More

Specify a width class for the SmMore component to best fit the text in the options.

        }

      template: `
        <div>
            <div class="ma2 pa2 w-50 flex flex-column ba">
                <SmMore class="mv1 w4" style="margin-left:auto;"

                    v-on:first="setEvent('first')"
                    v-on:second="event='second'"/>
                <SmMore  class="mv1 w5" style="margin-left:auto;"

                    v-on:edit="setEvent('edit')"
                    v-on:delete="event='delete'"
                    v-on:change_location="event='change_location'"/>
                <SmMore class="mv1 w-100"
                    :options="['Monday Tuesday Wednesday Thursday Friday Saturday Sunday', 'None']"
                    />
            </div>
            <div v-if="event" class="ma2">
                Event: {{ event }}
            </div>
        </div>
