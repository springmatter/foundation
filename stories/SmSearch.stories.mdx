# SmSearch

A searchbar with built in fuzzy finding using [Fuse.js](https://fusejs.io/).

<Props of={SmSearch} />

## Search over Strings

<Preview>
  <Story name="simple search" height="300px">
    {{
      components: { SmSearch },
      data() {
        return {
          results: [
            "Apple",
            "Pear",
            "Banana",
            "Mango",
            "Guava",
            "Lemon",
            "Grapefruit"
          ]
        };
      },
      template: `
          <div>
              <SmSearch style="width:300px" class="ma2"
                v-model="results" />
              <div class="ma2 mt3">
              Results:
                <div v-for="result in results" class="ma2">
                  {{result}}
                </div>
              </div>
          </div>
          `
    }}
  </Story>
</Preview>
