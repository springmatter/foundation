# SmSelect

A simple component for selecting an option from a dropdown list.

<Props of={SmSelect} />

## Simple Select

Lists with more than 10 single-lined options are scrollable.

<Preview>
  <Story name="simple" height="400px">
    {{
      components: { SmSelect },
      template: `
      <div class="flex">
        <SmSelect class="ma2" style="width:300px;flex-shrink:0" 
            :options="['Apple', 'Pear', 'Banana', 'Mango', 'Guava', 'Lemon', 'Grapefruit', 'Lime', 'Apricot', 'Orange', 'Plum']"/>
        <SmSelect class="ma2" style="width:300px;flex-shrink:0" 
            :options="['I am a really long and overflowing option', 'NA']"/>
       <SmSelect class="ma2" style="width:300px;flex-shrink:0" 
            displayKey="name"

                       { name: 'Jack', type: 'cat', color: 'black'}, 
                       { name: 'Pepper', type: 'dog', color: 'mixed'}, 
                       { name: 'Lucy', type: 'cat', color: 'white'}]" />
      </div>
      `
    }}
  </Story>
</Preview>
